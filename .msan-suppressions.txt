# MemorySanitizer suppressions for the CI memory-sanitizer job.
#
# wrp/omni are compiled with -fsanitize=memory, but the runner's libstdc++ is
# NOT instrumented. Any value that originates inside libstdc++ (std::string
# buffers from out-of-line operator+, the std::ctype widen cache, etc.) is seen
# as "uninitialized" by MSan, producing use-of-uninitialized-value false
# positives on routine calls such as:
#   * std::ifstream / std::__basic_file<char>::open  -> "Uninitialized ... in fopen"
#   * std::endl -> std::ctype<char>::_M_widen_init    -> "Uninitialized ... in memcmp"
#
# Suppress reports whose interceptor call stack passes through libstdc++ so the
# program is not aborted at the first such report and runs to its real exit
# code (matching the non-sanitizer builds, which pass these tests).
interceptor_via_lib:libstdc
